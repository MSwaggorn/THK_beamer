% -------------------- font sizes -------------------- %
% adjust these to change text sizes throughout the presentation
% syntax: \fontsize{size}{line spacing}\selectfont

% --- slide content fonts --- %
\newcommand{\headingsize}{\fontsize{12pt}{14pt}\selectfont}
\newcommand{\subheadingsize}{\fontsize{10pt}{12pt}\selectfont}
\newcommand{\normaltextsize}{\fontsize{9pt}{11pt}\selectfont}
\newcommand{\captionsizetext}{\fontsize{7pt}{8pt}\selectfont}
\newcommand{\footnotesmallsize}{\fontsize{5pt}{6pt}\selectfont}

% --- title page fonts --- %
\newcommand{\titlesize}{\fontsize{14pt}{16pt}\selectfont}
\newcommand{\subtitlesize}{\fontsize{12pt}{14pt}\selectfont}

% apply font sizes to beamer elements
\setbeamerfont{frametitle}{size=\headingsize}
\setbeamerfont{normal text}{size=\normaltextsize}
\setbeamerfont{footnote}{size=\footnotesmallsize}
\setbeamerfont{caption}{size=\captionsizetext}
\setbeamerfont{caption name}{size=\captionsizetext}
