% --- colored text boxes --- %
% automatic height box - adjusts height to fit text content
% usage: \coloredbox[options]{color}{text}
% options: border = add darker border | width = custom width (default: \textwidth)
% examples: \coloredbox{THRed}{text}
%           \coloredbox[border]{THRed}{text}
%           \coloredbox[width=0.5\textwidth]{THRed}{text}
%           \coloredbox[border,width=0.7\textwidth]{THRed}{text}
\makeatletter
\define@key{cbox}{border}[true]{\def\cbox@border{#1}}
\define@key{cbox}{width}{\def\cbox@width{#1}}
\newcommand{\coloredbox}[3][]{%
    \setkeys{cbox}{border=false,width=\textwidth,#1}%
    \par\noindent\hspace*{\fill}%
    \edef\tempa{true}%
    \ifx\cbox@border\tempa
        \setlength{\fboxrule}{1.2pt}\fcolorbox{#2!70!black}{#2}{\parbox{\dimexpr\cbox@width-2\fboxsep-2.4pt}{\centering\textcolor{white}{#3}}}%
    \else
        \colorbox{#2}{\parbox{\dimexpr\cbox@width-2\fboxsep}{\centering\textcolor{white}{#3}}}%
    \fi
    \hspace*{\fill}\par
}

% manual height box - specify vertical padding to control height
% usage: \coloredboxmanual[options]{color}{text}{padding}
% options: border = add darker border | width = custom width (default: \textwidth)
% padding: vertical space above and below text (e.g., 1cm, 0.5cm)
% examples: \coloredboxmanual{THRed}{text}{1cm}
%           \coloredboxmanual[border]{THRed}{text}{1cm}
%           \coloredboxmanual[width=0.5\textwidth]{THRed}{text}{0.5cm}
\define@key{cboxm}{border}[true]{\def\cboxm@border{#1}}
\define@key{cboxm}{width}{\def\cboxm@width{#1}}
\newcommand{\coloredboxmanual}[4][]{%
    \setkeys{cboxm}{border=false,width=\textwidth,#1}%
    \par\noindent\hspace*{\fill}%
    \edef\tempa{true}%
    \ifx\cboxm@border\tempa
        \setlength{\fboxrule}{1.2pt}\fcolorbox{#2!70!black}{#2}{\parbox{\dimexpr\cboxm@width-2\fboxsep-2.4pt}{\vspace{#4}\centering\textcolor{white}{#3}\vspace{#4}}}%
    \else
        \colorbox{#2}{\parbox{\dimexpr\cboxm@width-2\fboxsep}{\vspace{#4}\centering\textcolor{white}{#3}\vspace{#4}}}%
    \fi
    \hspace*{\fill}\par
}
\makeatother
